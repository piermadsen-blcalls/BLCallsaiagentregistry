BigQuery query
Change the two dates in the BETWEEN clause before running. For a weekly update, set start = 7 days ago, end = yesterday.
SELECT
  ivr_name,
  DATE(created_at, 'America/Denver') AS call_date,
  COALESCE(publisher_name, '(direct)') AS publisher_name,
  COALESCE(advertiser_name, '(direct)') AS advertiser_name,
  COUNT(*) AS raw_calls,
  COUNTIF(advertiser_payin > 0) AS paid_calls,
  SAFE_DIVIDE(COUNTIF(advertiser_payin > 0), COUNT(*)) AS conversion,
  AVG(IF(advertiser_payin > 0, ivr_duration, NULL)) AS avg_duration,
  SUM(advertiser_payin) AS revenue,
  SUM(advertiser_payin - publisher_payout) AS margin
FROM `ringpartner-exchange.phone_lead_transactions.phone_lead_transactions`
WHERE LOWER(ivr_name) LIKE 'ascnd%'
  AND DATE(created_at, 'America/Denver') BETWEEN '2026-04-10' AND '2026-04-16'
GROUP BY ivr_name, call_date, publisher_name, advertiser_name
ORDER BY call_date DESC, revenue DESC
